Sprites.js is simple canvas sprites animation library. No setTimeout() or setInterval(). Just rAF.

Read docs on Github
Input image

This image has no paddings. It contains 8 frames.

Sprite init 
var parrot = new Sprite({
    src: 'img/papuga.png',
    id: 'canvas',
    width: 160,
    height: 156,
    image_width: 1280,
    err: true
});

Note: Works only for one canvas.

.loop(fps, direction, n) 
parrot.loop(10)
parrot.loop(5, "backward")
//play loop forward 3 times
parrot.loop(8, null, 3)
.play(fps, from, to, n) 

//play 3 times from 5th to 8th frame at 5fps
parrot.play(5, 5, 8, 3);
.pause() 

parrot.pause();
.reset(5) 

//reset to 5th frame
parrot.reset(5);
.frame() 

alert(parrot.frame());
